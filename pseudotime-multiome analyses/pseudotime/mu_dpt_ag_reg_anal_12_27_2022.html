#add pseudotime binning (to be used in downstream analysis)
binner <- function(x) {
  if(x >= 0 & x < 0.1){
    return(1)
  }
  if (x > 0.1 & x < 0.2){
    return(2)
  }
  if (x > 0.2 & x < 0.3){
    return(3)
  }
  if (x > 0.3 & x < 0.4){
    return(4)
  }
  if (x > 0.4 & x < 0.5){
    return(5)
  }
  if (x > 0.5 & x < 0.6){
    return(6)
  }
  if (x > 0.6 & x < 0.7){
    return(7)
  }
  if (x > 0.7 & x < 0.8){
    return(8)
  }
  if (x > 0.8 & x < 0.9){
    return(9)
  }
  if (x > 0.9 & x <= 1){
    return(10)
  }
}
